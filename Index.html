        document.getElementById('generateBtn').addEventListener('click', async () => {

            const skillsInput = document.getElementById('skillsInput').value;

            const outputDiv = document.getElementById('output');

            const loadingIndicator = document.getElementById('loadingIndicator');


            if (!skillsInput.trim()) {

                outputDiv.innerHTML = `<p class="text-center text-red-500">Please enter your skills and interests to generate ideas.</p>`;

                return;

            }


            loadingIndicator.classList.remove('hidden');

            outputDiv.innerHTML = '';


            try {

                let chatHistory = [];

                const prompt = `Based on the article's message of creating something useful and helping others in a difficult job market, generate three specific and actionable project ideas for someone with the following skills and interests: "${skillsInput}". Each idea should be a short paragraph and include a title.`;

                chatHistory.push({ role: "user", parts: [{ text: prompt }] });

                const payload = { contents: chatHistory };

                const apiKey = ""

                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-05-20:generateContent?key=${apiKey}`;


                let retries = 0;

                const maxRetries = 5;

                const baseDelay = 1000;


                const response = await new Promise((resolve, reject) => {

                    const makeRequest = async () => {

                        try {

                            const res = await fetch(apiUrl, {

                                method: 'POST',

                                headers: { 'Content-Type': 'application/json' },

                                body: JSON.stringify(payload)

                            });

                            if (res.status === 429 && retries < maxRetries) {

                                retries++;

                                const delay = baseDelay * Math.pow(2, retries);

                                setTimeout(makeRequest, delay);

                            } else if (!res.ok) {

                                reject(new Error(`API error: ${res.statusText}`));

                            } else {

                                resolve(res);

                            }

                        } catch (e) {

                            reject(e);

                        }

                    };

                    makeRequest();

                });


                const result = await response.json();

                const text = result?.candidates?.[0]?.content?.parts?.[0]?.text;


                if (text) {

                    const formattedText = text.replace(/\n/g, '<br>');

                    outputDiv.innerHTML = `<p>${formattedText}</p>`;

                } else {

                    outputDiv.innerHTML = `<p class="text-center text-red-500">Sorry, I couldn't generate ideas. Please try a different prompt.</p>`;

                }

            } catch (error) {

                console.error("Gemini API call failed:", error);

                outputDiv.innerHTML = `<p class="text-center text-red-500">Sorry, something went wrong. Please check the console for more details.</p>`;

            } finally {

                loadingIndicator.classList.add('hidden');

            }

        });
